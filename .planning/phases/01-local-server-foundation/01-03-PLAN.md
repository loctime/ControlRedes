```javascript
'use strict';
const fs = require('fs');
const path = require('path');

const WATCH_DIR = path.resolve(__dirname, '..', 'nuevas-publicaciones');
const ARCHIVE_DIR = path.resolve(__dirname, '..', 'publicaciones-anteriores');

/**
 * Lee el caption para un archivo HTML.
 * Busca [nombre].caption.txt en la misma carpeta. Fallback: nombre sin extensión.
 * @param {string} filename - Nombre del archivo HTML (sin ruta)
 * @returns {string} Texto del caption
 */
function readCaption(filename) {
  const basename = filename.replace(/\.html$/i, '');
  const captionPath = path.join(WATCH_DIR, `${basename}.caption.txt`);
  if (fs.existsSync(captionPath)) {
    return fs.readFileSync(captionPath, 'utf8').trim();
  }
  return basename;
}

/**
 * Archiva un HTML publicado: lo mueve a publicaciones-anteriores/ y escribe .meta.json.
 * @param {string} filename - Nombre del archivo HTML (sin ruta)
 * @param {string[]} platforms - Plataformas publicadas, e.g. ['instagram', 'linkedin']
 * @param {'success'|'partial'|'failed'} status - Estado de la publicación
 * @param {{width:number, height:number, fps:number, codec:string, duration:number}} videoSpecs
 */
function markComplete(filename, platforms, status, videoSpecs) {
  const srcPath = path.join(WATCH_DIR, filename);
  if (!fs.existsSync(srcPath)) {
    throw new Error(`[lifecycle] Archivo no encontrado: ${srcPath}`);
  }

  // Asegurar que el directorio de archivo exista
  if (!fs.existsSync(ARCHIVE_DIR)) {
    fs.mkdirSync(ARCHIVE_DIR, { recursive: true });
  }

  const destPath = path.join(ARCHIVE_DIR, filename);
  fs.renameSync(srcPath, destPath);

  const basename = filename.replace(/\.html$/i, '');
  const metaPath = path.join(ARCHIVE_DIR, `${basename}.meta.json`);

  const meta = {
    date: new Date().toISOString(),
    platforms,
    status,
    videoSpecs: {
      width: videoSpecs.width,
      height: videoSpecs.height,
      fps: videoSpecs.fps,
      codec: videoSpecs.codec,
      duration: videoSpecs.duration,
    },
  };

  fs.writeFileSync(metaPath, JSON.stringify(meta, null, 2), 'utf8');
}

/**
 * Ejecuta el pipeline para una lista de archivos.
 * En Phase 1 simula el procesamiento — las fases 2-4 reemplazarán el cuerpo del loop.
 * Envía status updates por Telegram a lo largo del proceso.
 *
 * sendStatusFn se pasa como parámetro en lugar de importarse directamente para
 * desacoplar lifecycle.js de telegram.js y facilitar pruebas unitarias.
 *
 * @param {string[]} files - Lista de filenames a procesar (en orden)
 * @param {object} state - Estado compartido del pipeline
 * @param {Function} sendStatusFn - async (bot, chatId, text) => void — inyectada por el caller
 * @param {import('node-telegram-bot-api')} bot - Instancia del bot Telegram
 * @param {string} chatId - Chat ID destino para los mensajes
 */
async function runPipeline(files, state, sendStatusFn, bot, chatId) {
  try {
    for (const filename of files) {
      state.activeFile = filename;

      // TELE-02: status update — grabando
      await sendStatusFn(bot, chatId, `Grabando ${filename}...`);

      // Phase 2-4 hook point: aquí irá la lógica real de grabación.
      // Por ahora simulamos el paso de grabación.

      // ROADMAP Phase 1 Success Criteria 4: stubs de publicación por plataforma
      // Phase 4 reemplazará estos stubs con la automatización DOM real.
      await sendStatusFn(bot, chatId, 'Publicando en Instagram...');
      // Phase 4 hook: await publishInstagram(filename, caption);

      await sendStatusFn(bot, chatId, 'Publicando en LinkedIn...');
      // Phase 4 hook: await publishLinkedIn(filename, caption);

      // Archivar con specs de placeholder — Phase 3 enviará specs reales vía POST /api/video-ready
      const placeholderSpecs = {
        width: 1080,
        height: 1920,
        fps: 30,
        codec: 'h264',
        duration: 0,  // Phase 3 enviará la duración real
      };

      markComplete(filename, ['instagram', 'linkedin'], 'success', placeholderSpecs);

      state.pendingFiles = state.pendingFiles.filter(f => f !== filename);
    }

    // TELE-02: status update — completado
    await sendStatusFn(bot, chatId, 'Completado ✓');
  } catch (err) {
    console.error('[pipeline] Error durante pipeline:', err.message);
    await sendStatusFn(bot, chatId, `Error en pipeline: ${err.message}`);
  } finally {
    // Siempre resetear estado al terminar (éxito o falla)
    state.pipeline = 'idle';
    state.activeFile = null;
  }
}

module.exports = { readCaption, markComplete, runPipeline };
```

- `markComplete` usa `fs.renameSync` — funciona en el mismo volumen (ambas carpetas son subdirectorios del proyecto).
- El bloque `finally` garantiza que `state.pipeline` vuelve a `'idle'` incluso si el pipeline falla a mitad.

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Telegram msg → filesystem | Un atacante con acceso al bot podría intentar explotar el pipeline para mover archivos arbitrarios |
| server.js → lifecycle.js markComplete | filename viene originalmente del Telegram handler pero ha pasado por fs.readdirSync — solo puede ser un nombre de archivo real en WATCH_DIR |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-03-01 | Tampering | lifecycle.js — markComplete path construction | mitigate | Construir srcPath con `path.join(WATCH_DIR, filename)` donde filename viene de `fs.readdirSync(WATCH_DIR).filter(f => f.endsWith('.html'))` — solo archivos reales del directorio. No aceptar filenames arbitrarios de requests HTTP. |
| T-03-02 | Information Disclosure | .meta.json escrito en publicaciones-anteriores/ | accept | Los .meta.json solo contienen metadatos de publicación (fecha, plataformas, status, specs de video) — sin tokens, credenciales ni PII del usuario. |
| T-03-03 | Denial of Service | runPipeline consume el hilo si hay muchos archivos | accept | En uso normal < 10 archivos por pipeline. Node.js single thread no es problema para operaciones sync de FS en este volumen. |
| T-03-04 | Repudiation | No hay log de qué archivos se publicaron y cuándo | mitigate | El .meta.json con `date: ISO8601` provee auditoría por archivo. Adicionalmente, los mensajes de Telegram ("Grabando X...", "Completado") quedan en el historial del chat. |
| T-03-05 | Elevation of Privilege | Callback onTrigger pasado a initBot podría ser manipulado | accept | initBot y server.js son código local del servidor — no hay superficie de ataque externa para modificar el callback. |
</threat_model>
