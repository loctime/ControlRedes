```javascript
'use strict';
const TelegramBot = require('node-telegram-bot-api');
const fs = require('fs');
const path = require('path');

const WATCH_DIR = path.resolve(__dirname, '..', 'nuevas-publicaciones');
const COMMAND_TRIGGER = 'publica lo nuevo';

/**
 * Inicializa el bot de Telegram con long-polling.
 * @param {object} state - Estado compartido del pipeline (mutado directamente)
 * @returns {TelegramBot} instancia del bot para uso interno
 */
function initBot(state) {
  const token = process.env.BOT_TOKEN;
  if (!token) {
    console.error('[Telegram] BOT_TOKEN no configurado — bot desactivado');
    return null;
  }

  const bot = new TelegramBot(token, { polling: true });

  bot.on('message', async (msg) => {
    const fromChatId = String(msg.chat.id);
    const allowedChatId = String(state.chatId);

    // D-07: Ignorar mensajes de otros chats silenciosamente
    if (fromChatId !== allowedChatId) return;

    const text = (msg.text || '').trim().toLowerCase();

    // Solo procesar el comando exacto
    if (text !== COMMAND_TRIGGER) return;

    // D-10: Guardia — pipeline ya activo
    if (state.pipeline === 'active') {
      await bot.sendMessage(fromChatId, 'Ya hay un pipeline activo');
      return;
    }

    // D-08: Leer archivos al momento del comando (no tracking previo)
    const files = fs.readdirSync(WATCH_DIR)
      .filter(f => f.endsWith('.html'))
      .sort();

    // D-09: Guardia — carpeta vacía
    if (files.length === 0) {
      await bot.sendMessage(fromChatId, 'No hay archivos nuevos');
      return;
    }

    // Activar pipeline y reportar archivos encontrados
    state.pipeline = 'active';
    state.pendingFiles = [...files];
    state.activeFile = null;

    const lista = files.map((f, i) => `${i + 1}. ${f}`).join('\n');
    await bot.sendMessage(
      fromChatId,
      `Iniciando pipeline para ${files.length} archivo(s):\n${lista}`
    );
  });

  bot.on('polling_error', (err) => {
    console.error('[Telegram] Polling error:', err.message);
  });

  return bot;
}

/**
 * Envía un mensaje de status al chat configurado.
 * @param {TelegramBot} bot - Instancia del bot
 * @param {string} chatId - ID del chat destino
 * @param {string} text - Mensaje a enviar
 */
async function sendStatus(bot, chatId, text) {
  if (!bot || !chatId) return;
  try {
    await bot.sendMessage(String(chatId), text);
  } catch (err) {
    console.error('[Telegram] Error enviando status:', err.message);
  }
}

module.exports = { initBot, sendStatus };
```

NOTAS CRÍTICAS:
- El `COMMAND_TRIGGER` debe ser `'publica lo nuevo'` (minúsculas) y compararse contra `text.toLowerCase().trim()` — esto cubre mayúsculas y espacios extras.
- La función `sendStatus` captura errores internamente para no romper el pipeline si Telegram falla.
- `state` se muta directamente (referencia compartida con server.js) — no se retorna ni se copia.

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Telegram API → servidor | Mensajes llegan por long-polling; el contenido de msg.text no es confiable |
| CHAT_ID config → validación | Un atacante que conozca el BOT_TOKEN puede enviar mensajes desde cualquier chat |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-02-01 | Spoofing | telegram.js — validación de CHAT_ID | mitigate | Comparar `String(msg.chat.id)` con `String(state.chatId)` en cada mensaje. Cualquier chatId diferente es ignorado silenciosamente sin log (no revelar que el bot existe). |
| T-02-02 | Tampering | msg.text procesado como comando | mitigate | Solo procesar si `text === COMMAND_TRIGGER` después de `.trim().toLowerCase()`. Sin eval, sin shell, sin operaciones de FS con el contenido del mensaje. |
| T-02-03 | Information Disclosure | Respuesta del bot a chats no autorizados | mitigate | No responder a chats no autorizados (ni error, ni confirmación). Solo ignorar. |
| T-02-04 | Denial of Service | Flood de mensajes al bot | accept | Bot de uso personal con un único CHAT_ID. Sin rate limiting en Phase 1 — carga esperada < 1 msg/min. |
| T-02-05 | Elevation of Privilege | BOT_TOKEN expuesto en repo | mitigate | BOT_TOKEN en .env (gitignored). .env.example con placeholder. server/.gitignore incluye .env. |
</threat_model>
